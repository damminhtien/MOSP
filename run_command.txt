# Configure and build a fresh Release directory.
PREFIXES="$(brew --prefix boost);$(brew --prefix tbb)"
cmake -S . -B Release \
  -DCMAKE_BUILD_TYPE=Release \
  -DCMAKE_PREFIX_PATH="$PREFIXES"
cmake --build Release -j4

# Show the current canonical CLI.
./Release/main --help

# Legacy smoke benchmark path.
# Useful for quick local validation only.

# Run the config-driven benchmark harness.
python3 bench/scripts/run_benchmark.py --config bench/configs/timecap.yaml --suite-id local_timecap
python3 bench/scripts/run_benchmark.py --config bench/configs/completion.yaml --suite-id local_completion
python3 bench/scripts/run_benchmark.py --config bench/configs/scaling.yaml --suite-id local_scaling

# Aggregate raw suites into paper-style tables.
python3 bench/scripts/aggregate_results.py \
  --suite local_completion \
  --suite local_timecap \
  --suite local_scaling \
  --analysis-id local_analysis

# Render deterministic figures from the aggregate output.
python3 bench/scripts/plot_results.py \
  --input-dir bench/results/figures/local_analysis

# Read the benchmark methodology docs.
sed -n '1,200p' docs/benchmark-protocol.md
sed -n '1,200p' docs/benchmark-quickstart.md
sed -n '1,200p' docs/metrics-definition.md
sed -n '1,200p' docs/reproducibility.md

# Prepare output directories for canonical artifacts.
mkdir -p output/frontiers
mkdir -p output/traces
mkdir -p output/readme-bench/frontiers
mkdir -p output/readme-bench/traces

# Single-query LTMOA smoke run.
./Release/main \
  -m maps/NY-d.txt maps/NY-t.txt maps/NY-m.txt \
  -s 96000 \
  -t 38000 \
  -a LTMOA \
  --budget-sec 0.05 \
  --summary-output output/summary.csv \
  --frontier-output-dir output/frontiers \
  --trace-output-dir output/traces \
  --dataset-id nyc_3obj \
  --query-id single_smoke

# Single-query SOPMOA_relaxed smoke run with 4 threads.
./Release/main \
  -m maps/NY-d.txt maps/NY-t.txt maps/NY-m.txt \
  -s 96000 \
  -t 38000 \
  -a SOPMOA_relaxed \
  -n 4 \
  --budget-sec 0.05 \
  --summary-output output/summary.csv \
  --frontier-output-dir output/frontiers \
  --trace-output-dir output/traces \
  --dataset-id nyc_3obj \
  --query-id relaxed_smoke

# Scenario-mode smoke run.
./Release/main \
  -m maps/NY-d.txt maps/NY-t.txt maps/NY-m.txt \
  --scenario scen/query6.json \
  --from 0 \
  --to 1 \
  -a SOPMOA \
  -n 4 \
  --budget-sec 0.05 \
  --summary-output output/scenario-summary.csv \
  --frontier-output-dir output/frontiers \
  --trace-output-dir output/traces \
  --dataset-id nyc_3obj \
  --query-id scenario_smoke

# Small comparison sweep, including SOPMOA with 4 threads.
for ALG in SOPMOA SOPMOA_relaxed LTMOA LazyLTMOA LTMOA_array LazyLTMOA_array EMOA NWMOA; do
  THREAD_ARGS=()
  if [ "$ALG" = "SOPMOA" ] || [ "$ALG" = "SOPMOA_relaxed" ]; then
    THREAD_ARGS=(-n 4)
  fi

  ./Release/main \
    -m maps/NY-d.txt maps/NY-t.txt maps/NY-m.txt \
    -s 175000 \
    -t 189000 \
    -a "$ALG" \
    "${THREAD_ARGS[@]}" \
    --budget-sec 2 \
    --summary-output output/readme-bench/summary.csv \
    --frontier-output-dir output/readme-bench/frontiers \
    --trace-output-dir output/readme-bench/traces \
    --dataset-id nyc_3obj \
    --query-id "$ALG"
done

# Run the correctness harness.
./scripts/run_correctness_harness.sh Release

# Run the measurement smoke tests directly.
ctest --test-dir Release --output-on-failure -R benchmark_metrics_smoke
