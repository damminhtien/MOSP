# Build a fresh Release directory on macOS with Homebrew dependencies.
PREFIXES="$(brew --prefix boost);$(brew --prefix tbb);$(brew --prefix libomp)"
cmake -S . -B Release \
  -DCMAKE_BUILD_TYPE=Release \
  -DCMAKE_PREFIX_PATH="$PREFIXES" \
  -DOpenMP_ROOT="$(brew --prefix libomp)"
cmake --build Release -j

# Show the current CLI.
./Release/main --help

# Prepare local output directories.
mkdir -p output
mkdir -p output/bench
mkdir -p output/bench-wallclock
mkdir -p output/bench-suite
mkdir -p output/SOPMOA

# Single-query SOPMOA smoke test on the 3-objective NYC maps.
./Release/main \
  -m maps/NY-d.txt maps/NY-t.txt maps/NY-m.txt \
  -s 88959 \
  -t 96072 \
  -o output/sopmoa_smoke.csv \
  -a SOPMOA \
  --timelimit 2 \
  -n 4

# Single-query SOPMOA_relaxed smoke test on the same query.
./Release/main \
  -m maps/NY-d.txt maps/NY-t.txt maps/NY-m.txt \
  -s 88959 \
  -t 96072 \
  -o output/sopmoa_relaxed_smoke.csv \
  -a SOPMOA_relaxed \
  --timelimit 2 \
  -n 4

# Scenario-based SOPMOA run for the first query in scen/query3.json.
./Release/main \
  -m maps/NY-d.txt maps/NY-t.txt maps/NY-m.txt \
  --scenario scen/query3.json \
  --from 0 \
  --to 1 \
  -o output/sopmoa_query3.csv \
  --logsols output/SOPMOA \
  -a SOPMOA \
  --timelimit 2 \
  -n 4

# Canonical fair benchmark protocol used in README.md.
# This runs every solver with the same wall-clock cutoff, prints a 1-thread
# comparison table, then prints a separate 1-vs-4 scalability table for the
# parallel solvers.
python3 scripts/benchmark_wall_clock.py --runs 5 | tee output/bench-wallclock/benchmark.txt

# Longer and more diverse suite benchmark.
# This samples multiple scenario files and is better for cross-query comparisons.
python3 scripts/benchmark_suite.py \
  --scenario-files scen/query2.json scen/query3.json scen/query4.json scen/query5.json scen/query6.json \
  --queries-per-scenario 1 \
  --time-limit 5 \
  --runs 1 | tee output/bench-suite/full-suite.txt

# Focused head-to-head: SOPMOA_relaxed vs LTMOA_array on the same diverse suite.
python3 scripts/benchmark_suite.py \
  --algorithms SOPMOA_relaxed LTMOA_array \
  --scenario-files scen/query2.json scen/query3.json scen/query4.json scen/query5.json scen/query6.json \
  --queries-per-scenario 2 \
  --time-limit 5 \
  --runs 1 | tee output/bench-suite/relaxed-vs-ltmoa-array.txt

# Optional: one-thread-only throughput pass without the summary script.
for ALG in SOPMOA SOPMOA_relaxed LTMOA LazyLTMOA LTMOA_array LazyLTMOA_array EMOA NWMOA; do
  ./Release/main \
    -m maps/NY-d.txt maps/NY-t.txt maps/NY-m.txt \
    -s 88959 \
    -t 96072 \
    -o "output/bench-wallclock/${ALG}-t1.csv" \
    -a "$ALG" \
    --timelimit 2 \
    -n 1
done

# Optional: separate scalability pass for the parallel solvers.
for ALG in SOPMOA SOPMOA_relaxed; do
  for THREADS in 1 4; do
    ./Release/main \
      -m maps/NY-d.txt maps/NY-t.txt maps/NY-m.txt \
      -s 88959 \
      -t 96072 \
      -o "output/bench-wallclock/${ALG}-t${THREADS}.csv" \
      -a "$ALG" \
      --timelimit 2 \
      -n "$THREADS"
  done
done

# Optional: inspect the benchmark CSV files.
for CSV in output/bench-wallclock/*.csv; do
  echo "== $CSV =="
  cat "$CSV"
done
